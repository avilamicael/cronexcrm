```python
# products/models.py
from accounts.utils import BaseCompanyModel

class Product(BaseCompanyModel):
    """Produto vinculado automaticamente √† empresa"""
    name = models.CharField(max_length=200)
    description = models.TextField(blank=True)
    price = models.DecimalField(max_digits=10, decimal_places=2)
    is_active = models.BooleanField(default=True)
    
    # ‚úÖ company, created_at, updated_at j√° inclu√≠dos via BaseCompanyModel
    # ‚úÖ Manager personalizado j√° inclu√≠do
    
    class Meta:
        verbose_name = "Produto"
        unique_together = ['company', 'name']  # Nome √∫nico por empresa
    
    def __str__(self):
        return f"{self.name} - {self.company.name}"

class Sale(BaseCompanyModel):
    """Venda vinculada √† empresa"""
    product = models.ForeignKey(Product, on_delete=models.CASCADE)
    quantity = models.PositiveIntegerField()
    unit_price = models.DecimalField(max_digits=10, decimal_places=2)
    total = models.DecimalField(max_digits=10, decimal_places=2)
    
    def save(self, *args, **kwargs):
        # ‚úÖ Garante que produto e venda s√£o da mesma empresa
        if self.product.company != self.company:
            raise ValueError("Produto deve ser da mesma empresa")
        self.total = self.quantity * self.unit_price
        super().save(*args, **kwargs)
```
